/*
  Logical Or ||
  -- Null + Undefined + Any Falsy Value
  Nullish Coalescing Operator ??
  -- Null + Undefined
*/


let price=false;
console.log(`The price is : ${price ||  200}`);// to retrive all false value
console.log(`The price is :${price ?? 200}`)// retrive null aNd  undefind like varibale name   just

console.log(Boolean(100));
console.log(Boolean(-100));
console.log(Boolean(0));
console.log(Boolean(""));
console.log(Boolean(null));
