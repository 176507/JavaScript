  String Methods

  - lastIndexOf(Value [Mand], Start [Opt] Length)

let a="Elzero wep School";
console.log(a.indexOf("wep"));//  to give element and take the poistion of if By default search from index 0
console.log(a.indexOf("wep",8));//to give element and take the poistion of if By default search from index 8
console.log(a.lastIndexOf("wep"));//to give element and take the poistion of if By default search from end index
console.log(a.lastIndexOf("o"));
//if the boistion not found it will give me -1
console.log(a.slice(2,6))// to  cut specific index
console.log(a.slice(-5,-3));//to  cut specific index from the end
console.log(a.repeat(5));//  to make repeat  
console.log(a.split()); // split all the variable
console.log(a.split("")); // split eatch element in the variable  
console.log(a.split(" ")); // split  from spaces  . 
console.log(a.split(" ",2)); // split  from spaces  up to 2 just .
